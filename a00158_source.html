180  // rotate the vector around a given axis by a given a angle
181  Vector3<T> getRotated(const Vector3<T> &axis, T angle) const {
182  Vector3<T> u = axis / axis.getR();
183  T c = cos(angle);
184  T s = sin(angle);
185  Vector3<T> Rx(c + u.x * u.x * (1 - c), u.x * u.y * (1 - c) - u.z * s,
186  u.x * u.z * (1 - c) + u.y * s);
187  Vector3<T> Ry(u.y * u.x * (1 - c) + u.z * s, c + u.y * u.y * (1 - c),
188  u.y * u.z * (1 - c) - u.x * s);
189  Vector3<T> Rz(u.z * u.x * (1 - c) - u.y * s,
190  u.z * u.y * (1 - c) + u.x * s, c + u.z * u.z * (1 - c));
191  return Vector3<T>(dot(Rx), dot(Ry), dot(Rz));
192  }
