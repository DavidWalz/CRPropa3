169  T interpolate(const Vector3d &position) const {
170  // position on a unit grid
171  Vector3d r = (position - gridOrigin) / spacing;
172 
173  // indices of lower and upper neighbors
174  int ix, iX, iy, iY, iz, iZ;
175  if (reflective) {
176  reflectiveClamp(r.x, Nx, ix, iX);
177  reflectiveClamp(r.y, Ny, iy, iY);
178  reflectiveClamp(r.z, Nz, iz, iZ);
179  } else {
180  periodicClamp(r.x, Nx, ix, iX);
181  periodicClamp(r.y, Ny, iy, iY);
182  periodicClamp(r.z, Nz, iz, iZ);
183  }
184 
185  // linear fraction to lower and upper neighbors
186  double fx = r.x - floor(r.x);
187  double fX = 1 - fx;
188  double fy = r.y - floor(r.y);
189  double fY = 1 - fy;
190  double fz = r.z - floor(r.z);
191  double fZ = 1 - fz;
192 
193  // trilinear interpolation (see http://paulbourke.net/miscellaneous/interpolation)
194  T b(0.);
195  //V000 (1 - x) (1 - y) (1 - z) +
196  b += get(ix, iy, iz) * fX * fY * fZ;
197  //V100 x (1 - y) (1 - z) +
198  b += get(iX, iy, iz) * fx * fY * fZ;
199  //V010 (1 - x) y (1 - z) +
200  b += get(ix, iY, iz) * fX * fy * fZ;
201  //V001 (1 - x) (1 - y) z +
202  b += get(ix, iy, iZ) * fX * fY * fz;
203  //V101 x (1 - y) z +
204  b += get(iX, iy, iZ) * fx * fY * fz;
205  //V011 (1 - x) y z +
206  b += get(ix, iY, iZ) * fX * fy * fz;
207  //V110 x y (1 - z) +
208  b += get(iX, iY, iz) * fx * fy * fZ;
209  //V111 x y z
210  b += get(iX, iY, iZ) * fx * fy * fz;
